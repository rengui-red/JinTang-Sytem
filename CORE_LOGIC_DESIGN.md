def 监控网络流束(进程名, 数据包):



&#x20;   # 1. 提取流束特征

&#x20;   时间间隔 = 数据包.获取发送间隔()

&#x20;   数据包大小 = 数据包.大小

&#x20;   目标地址 = 数据包.目标IP



&#x20;   # 2. 识别“心跳”或“异常流束”

&#x20;   # 如果每 60 秒发一次，且大小固定，极大概率为远控心跳

&#x20;   if 时间间隔 == 60秒 and 数据包大小 < 100字节:

&#x20;       风险等级 = "高危"

&#x20;

&#x20;   # 如果是 DNS 请求但域名字符串特别长（DNS隧道）

&#x20;   elif 数据包.协议 == "DNS" and len(数据包.域名) > 50:

&#x20;       风险等级 = "高危"



&#x20;   # 3. 迅速反应

&#x20;   if 风险等级 == "高危":

&#x20;       # A. 本地阻断

&#x20;       系统.阻断连接(进程名, 目标地址)

&#x20;       日志.记录(f"发现恶意流束！已阻断 {进程名} 连接 {目标地址}")

&#x20;

&#x20;       # B. 上报云端

&#x20;       云端接口.上报威胁({

&#x20;           "特征": "固定心跳包",

&#x20;           "IP": 目标地址,

&#x20;           "样本": 数据包.指纹

&#x20;       })

&#x20;

&#x20;       # C. 触发全局防御

&#x20;       云端接口.请求更新黑名单()
